21std::vector<std::string> findFilesWithExtension(const std::wstring& folder, const std::wstring& extension) {
22 std::vector<std::string> files;
23 WIN32_FIND_DATAW findFileData;
24 HANDLE hFind = FindFirstFileW((folder + L"\\*" + extension).c_str(), &findFileData);
25 if (hFind != INVALID_HANDLE_VALUE) {

27 if (!(findFileData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)) {
28 std::wstring wideFilePath = folder + L"\\" + findFileData.cFileName;
29 std::string filePath(wideFilePath.begin(), wideFilePath.end());
30 files.push_back(filePath);
31 }
32 } while (FindNextFileW(hFind, &findFileData) != 0);
33 FindClose(hFind);
34 }

37 }

49int GetEncoderClsid(const WCHAR* format, CLSID* pClsid) {
50 UINT num = 0; // Number of image encoders
51 UINT size = 0; // Size of the image encoder array in bytes
52 // Get the size of the image encoder array
53 Gdiplus::GetImageEncodersSize(&num, &size);
54 if (size == 0)
55 return -1; // Failure
56 // Create a buffer to hold the encoder array
57 Gdiplus::ImageCodecInfo* imageCodecInfo = (Gdiplus::ImageCodecInfo*)(malloc(size));
58 if (imageCodecInfo == NULL)
59 return -1; // Failure
60 // Get the image encoder array
61 GetImageEncoders(num, size, imageCodecInfo);
62 // Find the CLSID for the specified format
63 for (UINT i = 0; i < num; ++i) {
64 if (wcscmp(imageCodecInfo[i].MimeType, format) == 0) {
65 *pClsid = imageCodecInfo[i].Clsid;
66 free(imageCodecInfo);
67 return i; // Success
68 }
69 }
70 free(imageCodecInfo);
71 return -1; // Failure
72}
