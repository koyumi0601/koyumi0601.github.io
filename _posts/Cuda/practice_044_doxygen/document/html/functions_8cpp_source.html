86double calculateIndexWisePosition(const std::vector<double>& src, double target)
87{
88 if (src.empty()) return -1;
89 int size = src.size();
90 bool isAscending = src.back() > src.front();
91 double minVal = isAscending ? src.front() : src.back();
92 double maxVal = isAscending ? src.back() : src.front();
93 double scale = (size - 1) / (maxVal - minVal);
94 double exactPosition;
95 if (isAscending) {
96 exactPosition = (target - src.front()) * scale;
97 }
98 else {
99 exactPosition = (src.front() - target) * scale;
100 }
101 if ((exactPosition < 0) || (exactPosition >= size))
102 {
103 exactPosition = -1;
104 }
105 return exactPosition;
106}
